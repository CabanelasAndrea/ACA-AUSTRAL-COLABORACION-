        // Sistema de puntuación
        function setScore(criterion, value) {
            scores[criterion] = value;

            // Actualizar UI de botones
            for (let i = 0; i <= 4; i++) {
                const btn = document.getElementById(`${criterion}-${i}`);
                if (btn) {
                    if (i === value) {
                        btn.classList.add('active');
                        btn.style.background = getColorForScore(value);
                        btn.style.color = '#000';
                        btn.style.fontWeight = 'bold';
                    } else {
                        btn.classList.remove('active');
                        btn.style.background = 'rgba(31, 41, 55, 1)';
                        btn.style.color = '';
                        btn.style.fontWeight = '';
                    }
                }
            }

            calculateResults();
        }

        function getColorForScore(score) {
            const colors = {
                0: '#EF4444', // Rojo
                1: '#F97316', // Naranja
                2: '#EAB308', // Amarillo
                3: '#84CC16', // Lima
                4: '#22C55E'  // Verde
            };
            return colors[score] || '#6B7280';
        }

        function calculateResults() {
            // Calcular promedios por categoría
            const categories = {
                pavimento: ['pav1', 'pav2', 'pav3'],
                geo: ['geo1', 'geo2', 'geo3'],
                inter: ['int1', 'int2'],
                lat: ['lat1', 'lat2']
            };

            let totalSum = 0;
            let totalCount = 0;

            for (const [cat, items] of Object.entries(categories)) {
                let catSum = 0;
                let catCount = 0;

                items.forEach(item => {
                    if (scores[item] !== null) {
                        catSum += scores[item];
                        catCount++;
                    }
                });

                const catAvg = catCount > 0 ? (catSum / catCount).toFixed(1) : '0.0';
                const catElement = document.getElementById(`cat-${cat}`);
                const barElement = document.getElementById(`bar-${cat}`);

                if (catElement) catElement.textContent = catAvg;
                if (barElement) barElement.style.width = `${(catAvg / 4) * 100}%`;

                totalSum += catSum;
                totalCount += catCount;
            }

            // Calcular total
            const totalAvg = totalCount > 0 ? (totalSum / totalCount).toFixed(1) : '0.0';
            const totalElement = document.getElementById('total-score');
            if (totalElement) totalElement.textContent = totalAvg;

            // Actualizar círculo de progreso
            const circle = document.getElementById('progress-circle');
            if (circle) {
                const circumference = 326.72;
                const offset = circumference - ((totalAvg / 4) * circumference);
                circle.style.strokeDashoffset = offset;
            }

            // Actualizar mensaje de resultado
            updateResultMessage(parseFloat(totalAvg), totalCount);

            // Generar recomendaciones si hay suficientes datos
            if (totalCount >= 5) {
                generateRecommendations();
            }
        }

        function updateResultMessage(score, count) {
            const msgDiv = document.getElementById('result-message');
            if (!msgDiv) return;

            let diagnosis = '';
            let colorClass = '';

            if (count === 0) {
                diagnosis = 'Complete la evaluación para obtener resultado';
                colorClass = 'text-gray-500';
            } else if (score < 1.5) {
                diagnosis = '⚠ CRÍTICO: Intervención inmediata requerida';
                colorClass = 'text-red-500';
            } else if (score < 2.5) {
                diagnosis = '⚠ REGULAR: Mejoras prioritarias necesarias';
                colorClass = 'text-yellow-500';
            } else if (score < 3.5) {
                diagnosis = '✓ ACEPTABLE: Optimizaciones recomendadas';
                colorClass = 'text-lime-500';
            } else {
                diagnosis = '✓ ÓPTIMO: Mantenimiento preventivo';
                colorClass = 'text-green-500';
            }

            msgDiv.innerHTML = `
                <div class="text-xs font-mono text-gray-500 mb-1">DIAGNÓSTICO (${count}/10 evaluados)</div>
                <div class="text-sm ${colorClass} font-bold">${diagnosis}</div>
            `;
        }

        function generateRecommendations() {
            const recommendations = [];

            // Analizar pavimento
            const pavScores = [scores.pav1, scores.pav2, scores.pav3].filter(s => s !== null);
            if (pavScores.length > 0) {
                const pavAvg = pavScores.reduce((a, b) => a + b, 0) / pavScores.length;
                if (pavAvg < 2) {
                    recommendations.push({
                        priority: 'high',
                        title: 'Rehabilitación de Pavimento',
                        desc: 'Implementar microaglomerado en zonas críticas y nivelación de tapas de servicios.',
                        timeframe: 'corto'
                    });
                } else if (pavAvg < 3) {
                    recommendations.push({
                        priority: 'medium',
                        title: 'Tratamiento Superficial',
                        desc: 'Aplicar riego de sellado y marcas viales antideslizantes.',
                        timeframe: 'medio'
                    });
                }
            }

            // Analizar geometría
            const geoScores = [scores.geo1, scores.geo2, scores.geo3].filter(s => s !== null);
            if (geoScores.length > 0) {
                const geoAvg = geoScores.reduce((a, b) => a + b, 0) / geoScores.length;
                if (geoAvg < 2) {
                    recommendations.push({
                        priority: 'high',
                        title: 'Rediseño Geométrico',
                        desc: 'Corregir radios de curva y peraltes inadecuados. Instalar señalización preventiva.',
                        timeframe: 'medio'
                    });
                }
            }

            // Analizar intersecciones
            const intScores = [scores.int1, scores.int2].filter(s => s !== null);
            if (intScores.length > 0) {
                const intAvg = intScores.reduce((a, b) => a + b, 0) / intScores.length;
                if (intAvg < 2) {
                    recommendations.push({
                        priority: 'high',
                        title: 'Intervención en Intersecciones',
                        desc: 'Instalar refugios peatonales, mejorar visibilidad y señalización vertical.',
                        timeframe: 'corto'
                    });
                }
            }

            // Analizar seguridad lateral
            const latScores = [scores.lat1, scores.lat2].filter(s => s !== null);
            if (latScores.length > 0) {
                const latAvg = latScores.reduce((a, b) => a + b, 0) / latScores.length;
                if (latAvg < 2) {
                    recommendations.push({
                        priority: 'medium',
                        title: 'Protección Lateral',
                        desc: 'Instalar barreras tipo motociclista o eliminar obstáculos rígidos.',
                        timeframe: 'medio'
                    });
                }
            }

            // Si todo está bien
            if (recommendations.length === 0) {
                recommendations.push({
                    priority: 'low',
                    title: 'Mantenimiento Preventivo',
                    desc: 'Continuar con inspecciones periódicas y conservación rutinaria.',
                    timeframe: 'largo'
                });
            }

            // Renderizar recomendaciones
            renderRecommendations(recommendations);
        }

        function renderRecommendations(recs) {
            const container = document.getElementById('recommendations-section');
            const list = document.getElementById('recommendations-list');

            if (!container || !list) return;

            const priorityLabels = {
                high: { text: 'ALTA', color: 'text-red-500', border: 'border-red-500' },
                medium: { text: 'MEDIA', color: 'text-yellow-500', border: 'border-yellow-500' },
                low: { text: 'BAJA', color: 'text-green-500', border: 'border-green-500' }
            };

            list.innerHTML = recs.map((rec, index) => `
                <div class="recommendation-card recommendation-animate priority-${rec.priority}" style="animation-delay: ${index * 0.1}s">
                    <div class="flex justify-between items-start mb-2">
                        <h4 class="text-lg font-bold text-white">${rec.title}</h4>
                        <span class="text-xs font-mono ${priorityLabels[rec.priority].color} border ${priorityLabels[rec.priority].border} px-2 py-1 rounded">
                            PRIORIDAD ${priorityLabels[rec.priority].text}
                        </span>
                    </div>
                    <p class="text-gray-400 text-sm mb-2">${rec.desc}</p>
                    <div class="text-xs text-cyan-400 font-mono">Plazo: ${getTimeframeText(rec.timeframe)}</div>
                </div>
            `).join('');

            // Actualizar prioridades de inversión
            updateInvestmentPriorities(recs);

            container.classList.remove('hidden');
            container.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        }

        function getTimeframeText(tf) {
            const texts = {
                corto: '0-6 meses',
                medio: '6-18 meses',
                largo: '18+ meses'
            };
            return texts[tf] || 'Por definir';
        }

        function updateInvestmentPriorities(recs) {
            const cortoEl = document.getElementById('prioridad-corto');
            const medioEl = document.getElementById('prioridad-medio');
            const largoEl = document.getElementById('prioridad-largo');

            const corto = recs.filter(r => r.timeframe === 'corto').map(r => r.title).join(', ') || 'Ninguna';
            const medio = recs.filter(r => r.timeframe === 'medio').map(r => r.title).join(', ') || 'Ninguna';
            const largo = recs.filter(r => r.timeframe === 'largo').map(r => r.title).join(', ') || 'Mantenimiento rutinario';

            if (cortoEl) cortoEl.textContent = corto;
            if (medioEl) medioEl.textContent = medio;
            if (largoEl) largoEl.textContent = largo;
        }
